main()
{
    a, prom : real;
    x, y : int;
    z, w : char;
    programDone : bool;
    fileRead : bool;

    programDone = false;
    fileRead = true;
    z = 'z';
    w = 'w';
    x = 123;
    y = $FF9AB12;
    prom = 0.5E10;
    
    while((programDone != false) || (fileRead && y > x))
    {
        z = z + 1;
        x = x - 3;
        y = y - w * x;
        a = y / prom - x;
    }
}
